  ```tsx
    'AS.DECIMAL':      '소수의 개념과 자릿값을 이해하는 능력',
    'AS.RATIO':        '비율과 비례 관계를 이해하는 능력',
  }

  const ALL_SKILL_IDS = Object.keys(SKILL_LABELS)
  ```

  **Component state:**
  ```tsx
  const { user } = useAuth()
  const [levels, setLevels] = useState<Record<string, number>>({})
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState(false)
  ```

  **useEffect** — fetch skill levels on mount, only when authenticated:
  ```tsx
  useEffect(() => {
    if (!user) return
    setLoading(true)
    setError(false)
    fetchSkillLevels()
      .then((data) => setLevels(data.levels ?? {}))
      .catch(() => setError(true))
      .finally(() => setLoading(false))
  }, [user])
  ```

  **Render — unauthenticated:**
  ```tsx
  if (!user) {
    return (
      <section>
        <h1>스킬 트리</h1>
        <p>
          <Link to={ROUTES.LOGIN}>로그인이 필요합니다</Link>
        </p>
      </section>
    )
  }
  ```

  **Render — loading:**
  ```tsx
  if (loading) {
    return (
      <section>
        <h1>스킬 트리</h1>
        <p>불러오는 중…</p>
      </section>
    )
  }
  ```

  **Render — error:**
  ```tsx
  if (error) {
    return (
      <section>
        <h1>스킬 트리</h1>
        <p>다시 시도해보세요</p>
      </section>
    )
  }
  ```

  **Render — grid of 7 skill cards:**
  ```tsx
  return (
    <section>
      <h1>스킬 트리</h1>
      <div className="skill-tree-grid">
        {ALL_SKILL_IDS.map((skillId) => {
          const level = levels[skillId] ?? 0
          return (
            <div key={skillId} className="skill-tree-card">
              <div className="skill-tree-card-label">
                {SKILL_LABELS[skillId]}
              </div>
              <div className="skill-level-bar">
                {[1, 2, 3].map((seg) => (
                  <div
                    key={seg}
                    className={`skill-level-segment${level >= seg ? ' filled' : ''}`}
                  />
                ))}
              </div>
              <div
                className={`skill-tree-card-level-text${level > 0 ? ' mastered' : ''}`}
              >
                {level > 0 ? `레벨 ${level} / 3` : '미학습'}
              </div>
              <div className="skill-tree-card-desc">
                {SKILL_DESCRIPTIONS[skillId]}
              </div>
            </div>
          )
        })}
      </div>
    </section>
  )
  ```
